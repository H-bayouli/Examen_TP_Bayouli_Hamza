int Findmax(int T[], int n){
  int i, max=T[0];
for(i=1;i<n;i++){
  if (T[i] > max)
    max=T[i];
    }
  return max;

}

int Findmin(int T[], int n){
  int i; min=T[0];
for(i=0; i< n; i++){
  if (T[i] < min)
    min=T[i];
    }
  return min;
}

void compare(int T[], int n){
   printf("Le maximum est %d \n", Findmax(T,n));
   printf("Le minimum est %d \n", Findmin(T,n));

}
